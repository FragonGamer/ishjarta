299  public static void Save()
300  {
301  var objects = GameObject.FindObjectsOfType(typeof(MonoBehaviour));
302 
303  for (int i = 0; i < objects.Length; i++)
304  {
305  var element = objects[i];
306 
307  //Debug.Log(element);
308 
309  if (element is Player p)
310  {
311  SaveData.Instance.playerData = new PlayerData(p);
312  }
313  else if (element is Enemy e)
314  {
315  SaveData.Instance.enemyData.Add(new EnemyData(e));
316  }
317  else if (element is ItemManager im)
318  {
319  Item item = im.GetItem();
320  Vector2 position = im.gameObject.transform.position;
321  if (item is MeleeWeapon mw)
322  {
323  SaveData.Instance.meleeWeaponData.Add(new MeleeWeaponData(mw, position));
324  }
325  else if (item is RangedWeapon re)
326  {
327  SaveData.Instance.rangedWeaponData.Add(new RangedWeaponData(re, position));
328  }
329  else if (item is ActiveItem ai)
330  {
331  SaveData.Instance.activeItemData.Add(new ActiveItemData(ai, position));
332  }
333  else if (item is PassiveItem pi)
334  {
335  SaveData.Instance.passivItemData.Add(new PassivItemData(pi, position));
336  }
337  else if (item is UsableItem ui)
338  {
339  SaveData.Instance.usableItemData.Add(new UsableItemData(ui, position));
340  }
341  }
342  }
343  SerializationManager.Save(fileName, SaveData.Instance);
344  }
